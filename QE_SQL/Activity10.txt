1. Orders issued against the salesman
 who works for customer ID 3007
SELECT *
FROM orders
WHERE salesman_id = (
    SELECT salesman_id
    FROM customers
    WHERE customer_id = 3007
);


2. Orders attributed to a salesman in New York
SELECT *
FROM orders
WHERE salesman_id IN (
    SELECT salesman_id
    FROM salesman
    WHERE city = 'New York'
);


 3. Count customers with grades above New York's average grade
SELECT COUNT(*) AS customer_count
FROM customers
WHERE grade > (
    SELECT AVG(grade)
    FROM customers
    WHERE city = 'New York'
);


 4. Orders handled by salesmen
 who earn the maximum commission
SELECT *
FROM orders
WHERE salesman_id IN (
    SELECT salesman_id
    FROM salesman
    WHERE commission = (
        SELECT MAX(commission)
        FROM salesman
    )
);
