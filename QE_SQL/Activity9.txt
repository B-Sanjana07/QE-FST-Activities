1. Which salesman is working for which customer
SELECT
    c.cust_name,
    s.name AS salesman_name
FROM customers c
JOIN salesman s
ON c.salesman_id = s.salesman_id;


-- 2. List of customers in ascending order

SELECT
    c.cust_name,
    c.grade,
    s.name AS salesman_name
FROM customers c
JOIN salesman s
ON c.salesman_id = s.salesman_id
WHERE c.grade < 300
ORDER BY c.cust_name ASC;


-- 3. Customers who appointed a salesman

SELECT
    c.cust_name,
    s.name AS salesman_name,
    s.commission
FROM customers c
JOIN salesman s
ON c.salesman_id = s.salesman_id
WHERE s.commission > 0.12;


-- 4. Order details:

SELECT
    o.ord_no,
    o.ord_date,
    o.purchase_amount,
    c.cust_name,
    s.name AS salesman_name,
    s.commission
FROM orders o
JOIN customers c
ON o.customer_id = c.customer_id
JOIN salesman s
ON o.salesman_id = s.salesman_id;
